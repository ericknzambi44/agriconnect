CREATE OR REPLACE FUNCTION public.fn_finalize_payment_on_delivery()
RETURNS TRIGGER AS $$
DECLARE
    v_montant_total DECIMAL(12,2);
    v_portefeuille_id UUID;
BEGIN
    -- Déclenchement uniquement lors du passage à 'LIVRE'
    IF (NEW.statut_expedition = 'LIVRE' AND (OLD.statut_expedition IS DISTINCT FROM 'LIVRE' OR OLD.statut_expedition IS NULL)) THEN
        
        -- 1. Assignation directe via sous-requête (plus robuste que SELECT INTO)
        v_montant_total := (
            SELECT prix_total_commande 
            FROM public.commande 
            WHERE id = NEW.commande_id 
            LIMIT 1
        );

        -- 2. Récupération de l'ID du portefeuille du vendeur
        v_portefeuille_id := (
            SELECT id 
            FROM public.portefeuille 
            WHERE user_id = NEW.vendeur_id 
            LIMIT 1
        );

        -- Vérification de sécurité pour éviter les calculs sur NULL
        IF v_montant_total IS NOT NULL AND v_portefeuille_id IS NOT NULL THEN
            
            -- 3. Mise à jour du solde
            UPDATE public.portefeuille
            SET 
                solde_bloque = solde_bloque - v_montant_total,
                solde_disponible = solde_disponible + v_montant_total,
                last_update = NOW()
            WHERE id = v_portefeuille_id;

            -- 4. Log de la transaction finale
            INSERT INTO public.transactions_portefeuille (
                portefeuille_id, 
                montant, 
                type, 
                statut, 
                reference_externe, 
                description
            ) VALUES (
                v_portefeuille_id, 
                v_montant_total, 
                'VENTE', 
                'COMPLETE', 
                NEW.id, 
                'Vente finalisée - Colis livré #' || NEW.id
            );
            
        END IF;

    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
