-- ==========================================
-- 1. FONCTION : CRÉATION DE L'EXPÉDITION (POST-PAIEMENT)
-- ==========================================
CREATE OR REPLACE FUNCTION public.fn_handle_paid_order()
RETURNS TRIGGER AS $$
DECLARE
    v_vendeur_id UUID;
    v_prod_id UUID;
    v_code_depot TEXT;
BEGIN
    -- Condition : le paiement vient de passer à 'PAYEE'
    IF (NEW.statut_paiement = 'PAYEE' AND (OLD.statut_paiement IS DISTINCT FROM 'PAYEE')) THEN

        -- 1. Récupération du vendeur et du produit (SYNTAXE CORRIGÉE ICI)
        v_vendeur_id := (SELECT user_id FROM public.annonce WHERE id = NEW.annonce_id LIMIT 1);
        v_prod_id := (SELECT prod_id FROM public.annonce WHERE id = NEW.annonce_id LIMIT 1);

        -- 2. Mise à jour du stock de l'annonce
        UPDATE public.annonce
        SET quantite_restante = quantite_restante - NEW.quantite_commandee,
            statut = CASE 
                WHEN (quantite_restante - NEW.quantite_commandee) <= 0 THEN 'vendu' 
                ELSE 'en_attente' 
            END
        WHERE id = NEW.annonce_id;

        -- 3. Mise à jour du stock global du produit
        UPDATE public.produit
        SET quantite_prod = quantite_prod - NEW.quantite_commandee
        WHERE id = v_prod_id;

        -- 4. Génération UNIQUE du Code Dépôt
        v_code_depot := upper(substring(md5(random()::text) from 1 for 6));

        -- 5. Création de l'expédition (code_retrait reste NULL)
        INSERT INTO public.expedition (
            commande_id,
            vendeur_id,
            acheteur_id,
            code_depot,
            code_retrait,
            statut_expedition,
            destination_ville,
            destination_details,
            id_agence_retrait
        ) VALUES (
            NEW.id,
            v_vendeur_id,
            NEW.acheteur_id,
            v_code_depot,
            NULL, 
            'A_DEPOSER',
            NEW.destination_ville,
            NEW.destination_details,
            NEW.id_agence_retrait
        );

        -- 6. Mise à jour du statut de la commande
        NEW.statut := 'validée';

    END IF;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- TRIGGER ATTACHÉ À LA COMMANDE
DROP TRIGGER IF EXISTS trg_after_payment_success ON public.commande;
CREATE TRIGGER trg_after_payment_success
AFTER INSERT OR UPDATE ON public.commande
FOR EACH ROW
EXECUTE FUNCTION public.fn_handle_paid_order();
